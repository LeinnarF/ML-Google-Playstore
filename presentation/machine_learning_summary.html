df['Success'] = (df['Installs'] >= 100_000).astype(int)

🔧 Data Preprocessing

1. Handling Missing & Categorical Data

Python 
# Handle 'Varies with device' in Size column
df['Size_varies'] = (df['Size'] == 'Varies with device').astype(int)
df['Size'] = pd.to_numeric(df['Size'], errors='coerce')

# Fill missing with category median, then overall median
df['Size'] = df.groupby('Category')['Size'].transform(
    lambda x: x.fillna(x.median())
)
df['Size'].fillna(df['Size'].median(), inplace=True)

K-Nearest Neighbors

Python 
from sklearn.neighbors import KNeighborsClassifier
from sklearn.model_selection import cross_val_score

# K-value optimization via cross-validation
best_k = 0
best_score = 0

for k in range(1, 51, 2):
    knn_temp = KNeighborsClassifier(n_neighbors=k)
    scores = cross_val_score(
        knn_temp, X_train, y_train, 
        cv=5, 
        scoring='f1_weighted'
    )
    mean_score = scores.mean()
    
    if mean_score > best_score:
        best_score = mean_score
        best_k = k

# Train final model with optimal K
knn_final = KNeighborsClassifier(n_neighbors=best_k)
knn_final.fit(X_train, y_train)

# Evaluate
y_pred_knn = knn_final.predict(X_test)
accuracy_knn = knn_final.score(X_test, y_test)
print(f'Optimal K: {best_k}')
print(f'Test Accuracy: {accuracy_knn:.2%}')

def predict_app_success(model, training_columns):
    print("\n" + "="*40)
    print("APP SUCCESS PREDICTOR (100k+ Installs)")
    print("="*40)

    # Collect user inputs
    category = input("Enter Category (e.g., GAME, BUSINESS): ")
    app_type = input("Enter Type (Free/Paid): ")
    price = float(input("Enter Price (0 for free): "))
    content_rating = input("Enter Content Rating (e.g., Everyone): ")
    size_varies = int(input("Size varies with device? (0=No, 1=Yes): "))
    size = float(input("Enter Size in KB (if known): "))
    rating = float(input("Expected Rating (1.0-5.0): "))
    reviews = int(input("Expected number of reviews: "))
    
    # Create input dataframe and encode
    input_data = pd.DataFrame([{
        'Category': category,
        'Size_varies': size_varies,
        'Size': size,
        'Type': app_type,
        'Price': price,
        'Content Rating': content_rating,
        'Rating': rating,
        'Reviews': reviews
    }])
    
    input_encoded = pd.get_dummies(input_data)
    input_encoded = input_encoded.reindex(columns=training_columns, fill_value=0)
    
    # Make prediction
    prediction = model.predict(input_encoded)[0]
    probability = model.predict_proba(input_encoded)[0][1]
    
    if prediction == 1:
        print(f"\n✅ Prediction: HIT (High Volume Expected)")
        print(f"Confidence: {probability * 100:.1f}%")
    else:
        print(f"\n❌ Prediction: MISS (Low Volume Expected)")
        print(f"Confidence: {(1 - probability) * 100:.1f}%")

# Run predictor
predict_app_success(knn_final, X_encoded.columns)
