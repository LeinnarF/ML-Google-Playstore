2. Computing Summary Statistics

Python 
# Calculate statistical measures
mean_stats = numeric_df.mean()
median_stats = numeric_df.median()
variance_stats = numeric_df.var()

# Create consolidated summary table
eda_summary = pd.DataFrame({
    "mean": mean_stats,
    "median": median_stats,
    "variance": variance_stats
})

# Sort by variance (highest variability first)
eda_summary = eda_summary.sort_values(by="variance", ascending=False)

Python - Category Analysis 
# Group by category and compute statistics
numeric_cols = df.select_dtypes(include="number").columns.tolist()

category_summary = df.groupby("Category")[numeric_cols].agg([
    "mean", "median", "var"
])

# Analyze specific metric by category
metric = "Rating"
cat_metric = (
    df.groupby("Category")[metric]
    .agg(mean="mean", median="median", variance="var", count="count")
    .sort_values("count", ascending=False)
)
